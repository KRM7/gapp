1528  /* Find all Pareto fronts in the population and also assign the nondomination ranks of the candidates. */
1529  static std::vector<std::vector<size_t>> nonDominatedSort(Population& pop)
1530  {
1531  using namespace std;
1532 
1533  /* Calc the number of candidates which dominate each candidate, and the indices of the candidates it dominates. */
1534  vector<size_t> dom_count(pop.size(), 0);
1535  vector<vector<size_t>> dom_list(pop.size());
1536 
1537  for (size_t i = 0; i < pop.size(); i++)
1538  {
1539  for (size_t j = 0; j < i; j++)
1540  {
1541  if (detail::paretoCompare(pop[j].fitness, pop[i].fitness))
1542  {
1543  dom_count[j]++;
1544  dom_list[i].push_back(j);
1545  }
1546  else if (detail::paretoCompare(pop[i].fitness, pop[j].fitness))
1547  {
1548  dom_count[i]++;
1549  dom_list[j].push_back(i);
1550  }
1551  }
1552  }
1553 
1554  /* Find the indices of all non-dominated candidates (first/best pareto front). */
1555  vector<size_t> front;
1556  for (size_t i = 0; i < pop.size(); i++)
1557  {
1558  if (dom_count[i] == 0)
1559  {
1560  front.push_back(i);
1561  pop[i].rank = 0;
1562  }
1563  }
1564  /* Find all the other pareto fronts. */
1565  vector<vector<size_t>> pareto_fronts;
1566  size_t front_idx = 1;
1567  while (!front.empty())
1568  {
1569  /* "Remove" the current front and find the next one. */
1570  vector<size_t> next_front;
1571  for (const auto& i : front)
1572  {
1573  for (const auto& j : dom_list[i])
1574  {
1575  /* j belongs to the next front if it's domination count will become 0. */
1576  if (--dom_count[j] == 0)
1577  {
1578  next_front.push_back(j);
1579  pop[j].rank = front_idx;
1580  }
1581  }
1582  }
1583  pareto_fronts.push_back(front);
1584  front = next_front;
1585  front_idx++;
1586  }
1587 
1588  return pareto_fronts;
1589  }

1591  /* Calculate the crowding distances of the candidates in each pareto front of the population. */
1592  static void calcCrowdingDistances(Population& pop, std::vector<std::vector<size_t>>&& pfronts)
1593  {
1594  using namespace std;
1595  assert(!pop.empty());
1596 
1597  for_each(execution::seq, pfronts.begin(), pfronts.end(),
1598  [&pop](vector<size_t>& pfront)
1599  {
1600  /* Calc the distances in each fitness dimension. */
1601  for (size_t d = 0; d < pop[0].fitness.size(); d++)
1602  {
1603  sort(pfront.begin(), pfront.end(),
1604  [&pop, &d](size_t lidx, size_t ridx)
1605  {
1606  return pop[lidx].fitness[d] < pop[ridx].fitness[d];
1607  });
1608 
1609  /* Calc the crowding distance for each solution. */
1610  double finterval = pop[pfront.back()].fitness[d] - pop[pfront.front()].fitness[d];
1611  finterval = max(finterval, 1E-6);
1612 
1613  for (const auto& idx : pfront)
1614  {
1615  pop[idx].distance = 0.0;

3809  static void standardMutate(Candidate& child, double pm, double ps, double pi, size_t base_)
3810  {
3811  assert(0.0 <= pm && pm <= 1.0);
3812  assert(0.0 <= ps && ps <= 1.0);
3813  assert(0.0 <= pi && pi <= 1.0);
3814  assert(base_ > 1);
3815 
3816  /* Calc number of mutated genes. */
3817  double mean = child.chromosome.size() * pm;
3818  double SD = child.chromosome.size() * pm * (1 - pm);
3819 
3820  size_t mutation_count = size_t(round(rng::generateRandomNorm(mean, SD)));
3821  mutation_count = std::clamp(mutation_count, size_t{ 0 }, child.chromosome.size());
3822 
3823  /* The child will (very likely) be changed, and will need to be evaluated. */
3824  if (mutation_count > 0) child.is_evaluated = false;
3825 
3826  /*
3827  * Change mutation_count number of random genes.
3828  * One gene may be changed multiple times, but it's uncommon for long chromosomes, and not important.
3829  */
3830  for (size_t i = 0; i < mutation_count; i++)
3831  {
3832  size_t idx = rng::generateRandomIdx(child.chromosome.size());
3833  child.chromosome[idx] = rng::generateRandomInt(size_t{ 0 }, base_ - 1);
3834  }
3835 
3836  /* Perform swap with ps probability. */
3837  if (rng::generateRandomDouble() <= ps)
3838  {
3839  /* r1 and r2 might be the same index, but its rare for long chromosomes. */
3840  size_t r1 = rng::generateRandomIdx(child.chromosome.size());
3841  size_t r2 = rng::generateRandomIdx(child.chromosome.size());

3845  }
3846 
3847  /* Perform inversion with pi probability. */
3848  if (rng::generateRandomDouble() <= pi)
3849  {
3850  /* Pick a random range of genes. The bounds of the range may be the same, but its rare for long chromosomes. */

3856
